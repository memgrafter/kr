# Add resolve_uri and display_path helpers

## Notes

**2026-06-24T17:48:01Z**

Two pure functions, no IO:

fn resolve_uri(kr_folder: &Path, uri: &str) -> PathBuf {
    let clean = uri.strip_prefix('file://').unwrap_or(uri);
    // If absolute (starts with /), return as-is for backward compat
    if clean.starts_with('/') { return PathBuf::from(clean); }
    // Expand ~ in URI before joining
    let expanded = expand_tilde_in_path(clean);
    kr_folder.parent().unwrap().join(expanded)
}

fn display_path(abs: &Path) -> String {
    let home = dirs::home_dir().unwrap();
    if let Ok(stripped) = abs.strip_prefix(&home) {
        format!("~/{}", stripped.display())
    } else {
        format!("{}", abs.display())
    }
}
